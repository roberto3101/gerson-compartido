El motor del backend componible

En orden de arranque: el servidor monta las rutas, el catálogo declara las capacidades, los helpers de ruta componen las URLs, el proveedor decide qué está habilitado, y el ejecutor genérico atiende todos los endpoints generados.

Go · entradaaplicaciones/api/interno/servidor/rutas.goPunto de entrada: arma el servidor HTTP y registra todas las rutas.

Propósito: es el ensamblador del servidor. Crea el enrutador, registra la ruta de salud, las rutas hechas a mano, las rutas generadas, y envuelve todo con el middleware de registro.

func RegistrarRutas(configuracion bootstrap.Configuracion, servicios bootstrap.Servicios) http.Handler {
  mux := http.NewServeMux()
  manejador := nuevoManejadorAPI(configuracion, servicios)

  mux.HandleFunc("/salud", manejador.salud)
  manejador.registrarRutasAPI(mux)
  manejador.registrarRutasGeneradas(mux)

  return middlewareRegistro(mux)
}

Aquí se ve la convivencia de los dos mundos: rutas a mano (capacidades personalizadas) y rutas generadas (capacidades estándar). Ambas se montan en el mismo mux.

Go · catálogoaplicaciones/api/interno/plataforma/componibilidad/rutas.goHelpers que componen las URLs (no se escriben "/api/v1/..." a mano).

Propósito: centraliza los prefijos globales (/api, versión v1, /app) y arma rutas uniendo segmentos. Nadie escribe /api/v1/... a mano: lo compone RutaAPI.

const (
	PrefijoAPIBase        = "/api"
	VersionAPIBase        = "v1"
	PrefijoNavegacionBase = "/app"
)

func RutaAPI(segmentos ...string) string {
	return unirRuta(append([]string{PrefijoAPIBase, VersionAPIBase}, segmentos...)...)
}

func unirRuta(segmentos ...string) string {
	partes := make([]string, 0, len(segmentos))
	for _, segmento := range segmentos {
		limpio := strings.Trim(segmento, "/")
		if limpio != "" { partes = append(partes, limpio) }
	}
	return "/" + strings.Join(partes, "/")
}

type ProveedorCapacidades struct {
	rutaArchivo            string
	mutex                  sync.RWMutex
	capacidadesHabilitadas map[string]bool
}

func (p *ProveedorCapacidades) EstaHabilitada(codigo string) bool {
	p.refrescarSiCambio()
	p.mutex.RLock(); defer p.mutex.RUnlock()
	habilitada, existe := p.capacidadesHabilitadas[codigo]
	if !existe { return false }
	return habilitada
}

func (p *ProveedorCapacidades) refrescarSiCambio() {

	contenido, _ := os.ReadFile(p.rutaArchivo)

}
Go · rutas a manoaplicaciones/api/interno/servidor/registro_rutas.goLas rutas de las capacidades personalizadas, y la protección por habilitación.

Propósito: declara las rutas escritas a mano (capacidades personalizadas como contratos, gobierno_central, produccion) y las envuelve con el gate de capacidad. Es el equivalente "manual" de las operaciones generadas.

type registroRuta struct {
	patron    string   // /api/v1/...
	codigo    string   // capacidad
	manejador http.HandlerFunc
}

func (m *manejadorAPI) registrosRutasAPI() []registroRuta {
	return []registroRuta{
		{
			patron:    componibilidad.RutaAPI(componibilidad.CodigoCapacidadGobiernoCentral, "periodos"),
			codigo:    componibilidad.CodigoCapacidadGobiernoCentral,
			manejador: m.listarPeriodos,
		},
		{ /* roles, permisos, contratos, produccion... */ },
	}
}

func (m *manejadorAPI) registrarRutasAPI(mux *http.ServeMux) {
	for _, ruta := range m.registrosRutasAPI() {
		mux.HandleFunc(ruta.patron, m.protegerRuta(ruta))
	}
}

func (m *manejadorAPI) protegerRuta(ruta registroRuta) http.HandlerFunc {
	if ruta.codigo == "" { return ruta.manejador }
	return m.envolverConCapacidad(ruta.codigo, ruta.manejador)
}
Go · gateaplicaciones/api/interno/servidor/plataforma.goEl "portero": bloquea una ruta si su capacidad no está habilitada.

Propósito: dos cosas — expone el endpoint que lista el estado de todas las capacidades, y define envolverConCapacidad, el envoltorio que protege cada ruta según la activación por empresa/despliegue.

func (m *manejadorAPI) listarCapacidades(respuesta http.ResponseWriter, solicitud *http.Request) {
	estados := m.servicios.ProveedorCapacidades.ListarEstados(componibilidad.RegistroCapacidadesBase())
	responderJSON(respuesta, http.StatusOK, map[string]any{"exito": true, "datos": estados})
}

func (m *manejadorAPI) envolverConCapacidad(codigo string, siguiente http.HandlerFunc) http.HandlerFunc {
	return func(respuesta http.ResponseWriter, solicitud *http.Request) {
		if !m.servicios.ProveedorCapacidades.EstaHabilitada(codigo) {
			responderError(respuesta, http.StatusNotFound, "CAPACIDAD_DESHABILITADA", "...")
			return
		}
		siguiente(respuesta, solicitud) // pasa
	}
}

Go · ❤ ejecutoraplicaciones/api/interno/servidor/operaciones_generadas_base.goEl corazón: UN solo ejecutor genérico atiende TODOS los endpoints generados.

Propósito: el motor escrito a mano una sola vez. Define la forma de una operación, registra cada ruta envuelta en su gate, y al llegar una petición resuelve los parámetros (cabecera/query/cuerpo), valida y llama al SP. Nadie escribe un handler por endpoint.

type definicionParametroOperacion struct {
	Nombre, NombreEntrada, Tipo string
	Requerido bool
}

func (m *manejadorAPI) registrarRutasGeneradas(mux *http.ServeMux) {
	for _, operacion := range operacionesGeneradas() {
		mux.HandleFunc(operacion.Patron, m.envolverConCapacidad(operacion.CodigoCapacidad,
			func(w http.ResponseWriter, r *http.Request){ m.ejecutarOperacionGenerada(operacion, w, r) }))
	}
}

func (m *manejadorAPI) ejecutarOperacionGenerada(operacion definicionOperacionGenerada, ...) {
	if solicitud.Method != operacion.MetodoHTTP { responderMetodoNoPermitido(respuesta); return }
	if ... != http.MethodGet { json.NewDecoder(solicitud.Body).Decode(&cuerpo) } // cuerpo
	for _, parametro := range operacion.Parametros {
		valor, presente, err := extraerValorOperacion(solicitud, cuerpo, parametro)
		if !presente {
			if parametro.Requerido { responderError(... "PARAMETRO_OBLIGATORIO" ...); return }
			argumentos = append(argumentos, nil)
			continue
		}
		argumentos = append(argumentos, valor)
	}
	if operacion.TipoInvocacion == "procedimiento" {
		m.servicios.EjecutorFunciones.EjecutarProcedimiento(operacion.Funcion, argumentos...)
		return
	}
	resultado, err := m.servicios.EjecutorFunciones.EjecutarJSON(operacion.Funcion, argumentos...)
	responderJSONCrudo(respuesta, http.StatusOK, resultado)
}

func extraerValorOperacion(...) {
	if parametro.Nombre == "p_id_empresa" { return obtenerCabeceraObligatoria(solicitud, "X-Empresa-Id") }
	if parametro.Nombre == "p_id_usuario" { ... "X-Usuario-Id" ... }
	if valorQuery := ...Query().Get(parametro.NombreEntrada); valorQuery != "" { return convertirValorParametro(valorQuery, parametro.Tipo) }
	if valorCuerpo, existe := cuerpo[parametro.NombreEntrada]; existe { return convertirValorParametro(valorCuerpo, parametro.Tipo) }
}

func convertirValorParametro(valor any, tipo string) {
	switch {
	case strings.Contains(tipoNormalizado, "INT"): return strconv.Atoi(texto)
	case strings.Contains(tipoNormalizado, "BOOL"): return strconv.ParseBool(texto)
	case strings.Contains(tipoNormalizado, "NUMERIC"): return strconv.ParseFloat(texto, 64)
	}
}

Esta es la pieza que hace componible al backend: cero handlers por SP. La firma del SQL (que el generador convierte en definicionOperacionGenerada) es todo lo que necesita este ejecutor.

Go · middlewareaplicaciones/api/interno/servidor/middleware.goLoguea cada petición: método, ruta y cuánto tardó.

Propósito: envoltorio que registra cada request. Mide el tiempo, deja pasar la petición al enrutador, y loguea el resultado. Es lo que envuelve a todo el mux en rutas.go.

func middlewareRegistro(siguiente http.Handler) http.Handler {
	return http.HandlerFunc(func(respuesta http.ResponseWriter, solicitud *http.Request) {
		inicio := time.Now()
		siguiente.ServeHTTP(respuesta, solicitud)
		log.Printf("%s %s %s", solicitud.Method, solicitud.URL.Path, time.Since(inicio))
	})
}
